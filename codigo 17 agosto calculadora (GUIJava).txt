package guijava;
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
/**
 *
 * @author Estudiante
 */
public class Ventana extends JFrame implements ActionListener{
    JLabel La, Lb, Lr; 
    JTextField ta, tb; 
    JButton bt, bt2, bt3, bt4; 
    
    Ventana(){
        setSize(200,300);
        setTitle("Sumador");
        setLayout(new FlowLayout());
        La=new JLabel("Número a: ");
        Lb=new JLabel("Número b: ");
        Lr=new JLabel("");
        ta=new JTextField(5);
        tb=new JTextField(5);
        bt=new JButton("Sumar");
        bt2=new JButton("Restar");
        bt3=new JButton("Multiplicar");
        bt4=new JButton("Dividir");
        
        bt.addActionListener(this);
        bt2.addActionListener(this);
         bt3.addActionListener(this);
          bt4.addActionListener(this);
        add(La);
        add(ta);
        add(Lb);
        add(tb);
                
        
        add(bt);
         add(bt2);
          add(bt3);
          add(bt4);
        add(Lr);
        
        setVisible(true);
        setDefaultCloseOperation(EXIT_ON_CLOSE);
        
       
        
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        System.out.println("OK");
        double a=Double.parseDouble(ta.getText());
        double b=Double.parseDouble(tb.getText());
        double r=0;
        
        if (e.getSource()==bt){
            r=a+b;
              Lr.setText(""+r);
        }
        else if(e.getSource()==bt2){
            r=a-b;
              Lr.setText(""+r);
        }
        else if(e.getSource()==bt3){
            r=a*b;
              Lr.setText(""+r);
        }
        
        else if(e.getSource()==bt4){
            if(b==0){
                Lr.setText("Error");
            }
            else {
                r=a/b;
                  Lr.setText(""+r);
            }
        }
        
      
    }
    
    
}
